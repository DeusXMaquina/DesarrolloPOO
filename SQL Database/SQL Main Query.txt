CREATE FUNCTION studentInformation (@idStudent INT)
RETURNS TABLE
AS
RETURN(
SELECT a.id, a.name, a.lastName, a.secondLastName, CONVERT(VARCHAR, age) + ' Años' AS age, c.career, (SELECT COUNT(idCourse) FROM curriculum p WHERE p.idCareer = a.idCareer
GROUP BY idCareer) AS courses FROM student a
INNER JOIN careers c ON c.id = a.idCareer
WHERE a.id = @idStudent)

-- Valor de courses --

CREATE FUNCTION gradedCourses (@idStudent INT)
RETURNS TABLE
AS
RETURN (
SELECT course, grade FROM courses
INNER JOIN grades ON grades.idCourse = courses.id
WHERE idStudent = @idStudent
)

SELECT * FROM studentInformation(1);
